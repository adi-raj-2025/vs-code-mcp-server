var Print_HeadersVar = require('../jsc/Print_Headers/Print_Headers');
describe('print_Headers suit', function () {
    beforeEach(function () {
        var Context = function () {
        };
        Context.prototype = {
            setVariable: function (propertName, propertyValue) {
                this[propertName] = propertyValue;
            },
            getVariable: function (propertyName) {
                return this[propertyName];
            }
        };
        context = new Context();
    });
 
    it('1. pendingResponse set to Failed when response is null', function () {
        context.setVariable('response', null);
        print_Headers();
        expect(context['pendingResponse']).toBe('Failed');
    });
 
    it('2. pendingResponse set to Failed when response is undefined', function () {
        context.setVariable('response', undefined);
        print_Headers();
        expect(context['pendingResponse']).toBe('Failed');
    });
 
    it('3. pendingResponse set to response value when response is an object', function () {
        var testObj = { key: 'value' };
        context.setVariable('response', testObj);
        print_Headers();
        expect(context['pendingResponse']).toBe(testObj);
    });
 
    it('4. pendingResponse not set when response is a string', function () {
        context.setVariable('response', 'test string');
        print_Headers();
        expect(context['pendingResponse']).toBeUndefined();
    });
 
    it('5. pendingResponse set to response value when response is a number', function () {
        context.setVariable('response', 42);
        print_Headers();
        expect(context['pendingResponse']).toBe(42);
    });
 
});
